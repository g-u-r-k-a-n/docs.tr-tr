## <a name="example"></a><span data-ttu-id="014c5-108">Örnek</span><span class="sxs-lookup"><span data-stu-id="014c5-108">Example</span></span>

{
    [CLSCompliant(false)]
    int M();   // CS3010

{
    public int M()

        return 1;
    }

    public static void Main()
    {
