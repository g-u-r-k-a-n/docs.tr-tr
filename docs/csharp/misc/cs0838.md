## <a name="example"></a><span data-ttu-id="cb63a-108">Örnek</span><span class="sxs-lookup"><span data-stu-id="cb63a-108">Example</span></span>  

 <span data-ttu-id="cb63a-109">Aşağıdaki örnek CS0838 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="cb63a-109">The following example generates CS0838:</span></span>

using System.Linq.Expressions;

{  
    class Test  
    {  
        static int Main()  
        {  
  
            Expression<Func<int[,]>> expr =  
                () => new int[2, 2] { { 1, 2 }, { 3, 4 } }; // CS0838

            int[,] nums = new int[2, 2] { { 1, 2 }, { 3, 4 } };  
            Expression<Func<int[,]>> expr2 = () => nums;
  
            return 1;  
        }  
    }  
}
