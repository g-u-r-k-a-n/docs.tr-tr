## <a name="example"></a><span data-ttu-id="533a9-110">Örnek</span><span class="sxs-lookup"><span data-stu-id="533a9-110">Example</span></span>

    public delegate void Del();

    class Test
    {
        static int Main()
        {
            bool b1 = (() => { }) is Del;   // CS0837
            bool b2 = delegate() { } is Del;// CS0837

        }
    }
}
