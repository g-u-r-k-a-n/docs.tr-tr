   public class A { }  
}  
```  
  
 <span data-ttu-id="29d8b-109">Ardından, bu dosyayı derle:</span><span class="sxs-lookup"><span data-stu-id="29d8b-109">Then, compile this file:</span></span>

{  
   public class A { }  
}

   {  
      Console.WriteLine(typeof(Util.A)); // CS0435  
   }

```
