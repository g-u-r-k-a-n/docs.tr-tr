## <a name="example"></a><span data-ttu-id="ddd02-107">Örnek</span><span class="sxs-lookup"><span data-stu-id="ddd02-107">Example</span></span>  

 <span data-ttu-id="ddd02-108">Aşağıdaki kod CS1952 üretir:</span><span class="sxs-lookup"><span data-stu-id="ddd02-108">The following code produces CS1952:</span></span>

using System.Linq.Expressions;  
  
class Test  
{  
    public static int M(__arglist)  
    {  
        return 1;  
    }  
  
    static int Main()

        Expression<Func<int, int>> f = x => Test.M(__arglist(x)); // CS1952  
        return 1;  
    }  
}
