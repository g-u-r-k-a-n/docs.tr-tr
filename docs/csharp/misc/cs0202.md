 <span data-ttu-id="bc2ba-105"><xref:System.Collections.IEnumerable.GetEnumerator%2A>Foreach deyimlerinin kullanımını etkinleştirmek için kullanılan bir işlev, bir işaretçi veya dizi döndüremez; bir Numaralandırıcı görevi gören bir sınıfın örneğini döndürmelidir.</span><span class="sxs-lookup"><span data-stu-id="bc2ba-105">A <xref:System.Collections.IEnumerable.GetEnumerator%2A> function, used to enable the use of foreach statements, cannot return a pointer or array; it must return an instance of a class that is able to act as an enumerator.</span></span> <span data-ttu-id="bc2ba-106">Bir numaralandırıcı olarak kullanılacak uygun gereksinimler ortak bir geçerli özellik ve ortak bir MoveNext yöntemi içerir.</span><span class="sxs-lookup"><span data-stu-id="bc2ba-106">The proper requirements to serve as an enumerator include a public Current property and a public MoveNext method.</span></span>

```csharp  
// CS0202.cs  
  
public class C1  
{  
   public int Current  
   {  
      get  
      {  
         return 0;  
      }  
   }  
  
   public bool MoveNext ()  
   {  
      return false;  
   }  
  
   public static implicit operator C1 (int c1)  
   {  
      return 0;  
   }

public class C2  
{  
   public int Current  
   {  
      get  
      {  
         return 0;  
      }  
   }  
  
   public bool MoveNext ()  
   {  
      return false;  
   }  
  
   public C1[] GetEnumerator ()  
   // try the following line instead  
   // public C1 GetEnumerator ()  
   {  
      return null;  
   }  
}  
  
public class MainClass  
{  
   public static void Main ()  
   {  
      C2 c2 = new C2();  
  
      foreach (C1 x in c2)   // CS0202  
      {  
         System.Console.WriteLine(x.Current);  
      }  
   }  
}  
```
