## <a name="example"></a><span data-ttu-id="6ba30-108">Örnek</span><span class="sxs-lookup"><span data-stu-id="6ba30-108">Example</span></span>  
 <span data-ttu-id="6ba30-109">Aşağıdaki örnek CS0229 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="6ba30-109">The following example generates CS0229:</span></span>  
  
```csharp  
// CS0229.cs  
  
interface IList  
{  
    int Count  
    {  
        get;  
        set;  
    }  
  
    void Counter();  
}  
  
interface ICounter  
{  
    double Count  
    {  
        get;  
        set;  
    }  
}  
  
interface IListCounter : IList, ICounter {}  
  
class MyClass  
{  
    void Test(IListCounter x)  
    {  
        x.Count = 1;  // CS0229  
        // Try one of the following lines instead:  
        // ((IList)x).Count = 1;  
        // or  
        // ((ICounter)x).Count = 1;  
    }  
  
    public static void Main() {}  
}  
```
