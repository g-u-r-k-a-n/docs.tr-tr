 <span data-ttu-id="d0ab0-108">Aşağıdaki örnek CS0431 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="d0ab0-108">The following sample generates CS0431:</span></span>  
  
```csharp  
// CS0431.cs  
using A = Outer;  
  
public class Outer
{  
   public class Inner
   {  
      public static void Meth() {}  
   }  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      A::Inner.Meth();   // CS0431  
      A.Inner.Meth();   // OK  
   }  
}  
```
