## <a name="example"></a><span data-ttu-id="5ad83-107">Örnek</span><span class="sxs-lookup"><span data-stu-id="5ad83-107">Example</span></span>

```csharp  
{  
    public int i  
    {  
        set  
        {  
        }  
        // uncomment the get method to resolve this error  
        /*  
        get  
        {  
            return 0;  
        }

{  
    public static void Main()  
    {  
        MyClass2 myClass2 = new MyClass2();  
        int j = myClass2.i;   // CS0154, no get method  
    }  
}  
```
