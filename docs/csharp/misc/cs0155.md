 <span data-ttu-id="5f83d-105">**System.  <span data-ttu-id="5f83d-108">Aşağıdaki örnek CS0155 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="5f83d-108">The following sample generates CS0155:</span></span>  
  
```csharp  
// CS0155.cs  
using System;  
  
namespace MyNamespace  
{  
    public class MyClass2  
    // try the following line instead  
    // public class MyClass2 : Exception  
    {  
    }  
    public class MyClass  
    {  
        public static void Main()  
        {  
            try  
            {  
            }  
            catch (MyClass2)   // CS0155, resolves if you derive MyClass2 from Exception  
            {  
            }  
        }  
    }  
}  
```
