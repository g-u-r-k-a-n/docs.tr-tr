 <span data-ttu-id="34b75-108">Aşağıdaki örnek CS0030 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="34b75-108">The following sample generates CS0030:</span></span>  
  
```csharp  
// CS0030.cs  
namespace x  
{  
   public class iii  
   {  
      /*  
      public static implicit operator iii(int aa)  
      {  
         return null;  
      }  
  
      public static implicit operator int(iii aa)  
      {  
         return 0;  
      }  
      */  
  
      public static iii operator ++(iii aa)  
      {  
         return (iii)0;   // CS0030  
         // uncomment the conversion routines to resolve CS0030  
      }  
  
      public static void Main()  
      {  
      }  
   }  
}  
```
